# Commande pour compiler et lancer nativement sur ARM64 (Surface Pro 12)
# Assurez-vous d'avoir installé la cible rustup : rustup target add aarch64-pc-windows-msvc
# Et d'avoir les libs OpenCV pour ARM64.

$env:OPENCV_LINK_LIBS="opencv_world4130"; 
$env:OPENCV_LINK_PATHS="C:\tools\opencv_arm64\build\lib"; 
$env:OPENCV_INCLUDE_PATHS="C:\tools\opencv_arm64\build\include"; 
$env:CXX="clang-cl"; 
$env:CXXFLAGS="--target=aarch64-pc-windows-msvc"; 
$env:Path += ";C:\tools\opencv_arm64\build\bin"; 

# Utilisation du NPU via QNN/DirectML
cargo run --target aarch64-pc-windows-msvc --release -- .\video.mp4 --model model/scrfd_500m_bnkps.onnx --confidence 0.5 --fps
